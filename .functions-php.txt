function kommo_get_valid_token() {
    $expires = get_option('kommo_token_expires', 0);
    if ($expires > 0 && time() < $expires - 300) {
        return get_option('kommo_access_token');
    }
    $refresh_token = get_option('kommo_refresh_token');
    if (!$refresh_token) return false;

    $response = wp_remote_post(KOMMO_BASE_URL . '/oauth2/access_token', [
        'body' => json_encode([
            'client_id'     => KOMMO_CLIENT_ID,
            'client_secret' => KOMMO_CLIENT_SECRET,
            'grant_type'    => 'refresh_token',
            'refresh_token' => $refresh_token,
            'redirect_uri'  => 'https://passoapassouniformes.com'
        ]),
        'headers' => ['Content-Type' => 'application/json'],
        'timeout' => 30
    ]);

    if (is_wp_error($response)) return false;
    $data = json_decode(wp_remote_retrieve_body($response), true);
    if (empty($data['access_token'])) return false;

    update_option('kommo_access_token', $data['access_token']);
    update_option('kommo_refresh_token', $data['refresh_token']);
    update_option('kommo_token_expires', time() + $data['expires_in']);
    return $data['access_token'];
}

function kommo_criar_lead_pedido($order_id) {
    if (get_post_meta($order_id, '_kommo_lead_criado', true)) return;
    $order = wc_get_order($order_id);
    if (!$order) return;
    $token = kommo_get_valid_token();
    if (!$token) return;

    $nome  = $order->get_billing_first_name() . ' ' . $order->get_billing_last_name();
    $email = $order->get_billing_email();
    $fone  = $order->get_billing_phone();
    $total = $order->get_total();
    $items = [];
    foreach ($order->get_items() as $item) {
        $items[] = $item->get_name() . ' x' . $item->get_quantity();
    }
    $produtos = implode(', ', $items);

    $lead_data = [[
        'name'        => 'Pedido #' . $order_id . ' - ' . $nome,
        'price'       => (int)($total * 100),
        'pipeline_id' => KOMMO_PIPELINE_ID,
        'status_id'   => KOMMO_STATUS_ID,
        '_embedded'   => [
            'contacts' => [[
                'name'       => $nome,
                'first_name' => $order->get_billing_first_name(),
                'last_name'  => $order->get_billing_last_name(),
                'custom_fields_values' => array_values(array_filter([
                    $email ? ['field_code' => 'EMAIL', 'values' => [['value' => $email, 'enum_code' => 'WORK']]] : null,
                    $fone  ? ['field_code' => 'PHONE', 'values' => [['value' => $fone,  'enum_code' => 'WORK']]] : null,
                ]))
            ]]
        ],
        'custom_fields_values' => [[
            'field_name' => 'Produtos',
            'values'     => [['value' => $produtos]]
        ]]
    ]];

    $response = wp_remote_post(KOMMO_BASE_URL . '/api/v4/leads/complex', [
        'body'    => json_encode($lead_data),
        'headers' => [
            'Content-Type'  => 'application/json',
            'Authorization' => 'Bearer ' . $token
        ],
        'timeout' => 30
    ]);

    if (!is_wp_error($response) && wp_remote_retrieve_response_code($response) === 200) {
        update_post_meta($order_id, '_kommo_lead_criado', true);
    }
}

add_action('woocommerce_order_status_completed',  'kommo_criar_lead_pedido');
add_action('woocommerce_order_status_processing', 'kommo_criar_lead_pedido');
